285  template <typename T>
286  void addsource_(size_t i1,size_t i2,int oversample,
287  T &source,
288  PosType &total){
289  PosType tmp_res = resolution*1.0/oversample;
290  PosType tmp = tmp_res*tmp_res;
291  PosType bl = resolution /2 - 0.5*tmp_res;
292  PosType y[2],x[2];
293 
294  total = 0;
295 
296  for(size_t index = i1 ;index <= i2; ++index){
297  find_position(y,index);
298  y[0] -= bl;
299  y[1] -= bl;
300  for(int i = 0 ; i < oversample ; ++i){
301  x[0] = y[0] + i*tmp_res;
302  for(int j=0; j < oversample;++j){
303  x[1] = y[1] + j*tmp_res;
304  map[index] += source.SurfaceBrightness(x)*tmp;
305  total += source.SurfaceBrightness(x)*tmp;
306  }
307  }
308  }
