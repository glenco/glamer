311  template <typename T>
312  void addsource_(size_t i1,size_t i2,int oversample,
313  T &source,
314  PosType &total){
315  PosType tmp_res = resolution*1.0/oversample;
316  PosType tmp = tmp_res*tmp_res;
317  PosType bl = resolution /2 - 0.5*tmp_res;
318  PosType y[2],x[2];
319 
320  total = 0;
321 
322  for(size_t index = i1 ;index <= i2; ++index){
323  find_position(y,index);
324  y[0] -= bl;
325  y[1] -= bl;
326  for(int i = 0 ; i < oversample ; ++i){
327  x[0] = y[0] + i*tmp_res;
328  for(int j=0; j < oversample;++j){
329  x[1] = y[1] + j*tmp_res;
330  map[index] += source.SurfaceBrightness(x)*tmp;
331  total += source.SurfaceBrightness(x)*tmp;
332  }
333  }
334  }
335  }
