381  template <typename T>
382  void addsource_(size_t i1,size_t i2,int oversample,
383  T &source,
384  PosType &total){
385  PosType tmp_res = resolution*1.0/oversample;
386  PosType tmp = tmp_res*tmp_res;
387  PosType bl = resolution /2 - 0.5*tmp_res;
388  PosType y[2],x[2];
389 
390  total = 0;
391 
392  for(size_t index = i1 ;index <= i2; ++index){
393  find_position(y,index);
394  y[0] -= bl;
395  y[1] -= bl;
396  for(int i = 0 ; i < oversample ; ++i){
397  x[0] = y[0] + i*tmp_res;
398  for(int j=0; j < oversample;++j){
399  x[1] = y[1] + j*tmp_res;
400  map[index] += source.SurfaceBrightness(x)*tmp;
401  total += source.SurfaceBrightness(x)*tmp;
402  }
403  }
404  }
405  }
