346  template<typename T>
347  int cutbox(const T* center,PosType *p1,PosType *p2,float rmax){
348  /* returns: 0 if whole box is outside rmax from ray[]
349  * 1 if whole box is inside circle but ray is not in the box
350  * 2 if ray[] is inside box
351  * 3 if box intersects circle but ray[] is not inside box
352  */
353  short i,tick=0;
354  PosType close[2],rtmp;
355  PosType tmp1,tmp2;
356 
357  // find closest point on box borders to ray[]
358  for(i=0;i<2;++i){
359  if( center[i] < p1[i] ){
360  close[i]=p1[i];
361  }else if(center[i] > p2[i]){
362  close[i]=p2[i];
363  }else{
364  close[i]=center[i];
365  ++tick;
366  }
367  }
368 
369  if(tick==2) return 2; // ray is inside box
370 
371  for(i=0,rtmp=0;i<2;++i) rtmp += pow(center[i] - close[i],2);
372 
373  if(rtmp>rmax*rmax) return 0; // box is all outside circle
374 
375  // find farthest point on box border from ray[]
376  for(i=0,rtmp=0;i<2;++i) rtmp += ((tmp1 = pow(center[i]-p1[i],2)) > (tmp2=pow(center[i]-p2[i],2))) ? tmp1 : tmp2;
377  //for(i=0,rtmp=0;i<2;++i) rtmp += DMAX(pow(ray[i]-p1[i],2),pow(ray[i]-p2[i],2));
378 
379  if(rtmp<rmax*rmax) return 1; // box is all inside circle
380 
381  return 3; // box intersects circle
382  }
