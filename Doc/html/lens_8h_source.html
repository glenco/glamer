741  struct MINyFunction{
742  MINyFunction(Lens &mylens,Point_2d y,int sign):lens(mylens),y(y),sign(sign),r2max(0){}
743 
744  double operator()(double *x){
745  point.x[0] = x[1];
746  point.x[1] = x[2];
747  lens.rayshooterInternal(1,&point);
748  double r2 = (y[0]-point.image->x[0])*(y[0]-point.image->x[0])
749  + (y[1]-point.image->x[1])*(y[1]-point.image->x[1]);
750 
751  r2max = MAX(r2,r2max);
752  return r2 + r2max*abs(sign - sgn(point.invmag));
753  }
