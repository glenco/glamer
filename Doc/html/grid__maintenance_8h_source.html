248  bool EntirelyinCausticCurve(Point_2d x, PosType sourceRadius)
249  {
250  // Testing if the center of the source is within the caustic :
251  bool IsInCurve = Utilities::incurve(x.x,caustic_curve_outline);
252 
253  // Testing now that it is not too close from it (i.e. farther than source radius) :
254  int i=0; // index going over the different points of the caustic curve
255  PosType DistSourceToCautic; // distance between the source center and the considered point of the caustic line.
256 
257  if(IsInCurve == true) // center inside the caustic
258  {
259  while(i<caustic_curve_outline.size()) // testing that the source size does not overlap with the caustic line.
260  {
261  DistSourceToCautic = sqrt((caustic_curve_outline[i].x[0] - x.x[0])*(caustic_curve_outline[i].x[0] - x.x[0]) + (caustic_curve_outline[i].x[1] - x.x[1])*(caustic_curve_outline[i].x[1] - x.x[1]));
262  if (DistSourceToCautic < sourceRadius) return false ; // source too close, we return false and don't consider the point.
263  i++;
264  }
265  return true ; // if not we return true (the source is valid)
266  }
267  else return false ; // center not inside the caustic
268  }

315  void CriticalRadius(PosType &rmax,PosType &rmin,PosType &rave) const{
316  if(critical_curve.size() < 2){
317  rave = rmax = rmin = 0.0;
318  return;
319  }
320 
321  rave = rmin = rmax = (critical_center - critical_curve[0]).length();
322  PosType rad;
323 
324  for(size_t ii=1; ii< critical_curve.size(); ++ii){
325  rad = (critical_center - critical_curve[ii]).length();
326 
327  rave += rad;
328  if(rad < rmin) rmin = rad;
329  if(rad > rmax) rmax = rad;
330  }
331 
332  rave /= critical_curve.size();
333  }
