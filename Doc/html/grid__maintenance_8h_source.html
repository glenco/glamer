331  bool EntirelyinCausticCurve(Point_2d &x, PosType sourceRadius)
332  {
333  // Testing if the center of the source is within the caustic :
334  bool IsInCurve = Utilities::inCurve(x,caustic_curve_outline);
335 
336  // Testing now that it is not too close from it (i.e. farther than source radius) :
337  int i=0; // index going over the different points of the caustic curve
338  PosType DistSourceToCautic; // distance between the source center and the considered point of the caustic line.
339 
340  if(IsInCurve == true) // center inside the caustic
341  {
342  while(i<caustic_curve_outline.size()) // testing that the source size does not overlap with the caustic line.
343  {
344  DistSourceToCautic = sqrt((caustic_curve_outline[i].x[0] - x.x[0])*(caustic_curve_outline[i].x[0] - x.x[0]) + (caustic_curve_outline[i].x[1] - x.x[1])*(caustic_curve_outline[i].x[1] - x.x[1]));
345  if (DistSourceToCautic < sourceRadius) return false ; // source too close, we return false and don't consider the point.
346  i++;
347  }
348  return true ; // if not we return true (the source is valid)
349  }
350  else return false ; // center not inside the caustic
351  }

398  void CriticalRadius(PosType &rmax,PosType &rmin,PosType &rave) const{
399  if(critcurve.size() < 2){
400  rave = rmax = rmin = 0.0;
401  return;
402  }
403 
404  rave = rmin = rmax = (critical_center - critcurve[0].x).length();
405  PosType rad;
406 
407  for(size_t ii=1; ii< critcurve.size(); ++ii){
408  rad = (critical_center - critcurve[ii].x).length();
409 
410  rave += rad;
411  if(rad < rmin) rmin = rad;
412  if(rad > rmax) rmax = rad;
413  }
414 
415  rave /= critcurve.size();
416  }
