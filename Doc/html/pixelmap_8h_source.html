165 PosType AddSource_parallel(S &source,int oversample){
166 Point_2d s_center;
167 source.getTheta(s_center);
168
169 if( (s_center[0] + source.getRadius() ) < map_boundary_p1[0] ) return 0.0;
170 if( (s_center[0] - source.getRadius() ) > map_boundary_p2[0] ) return 0.0;
171 if( (s_center[1] + source.getRadius() ) < map_boundary_p1[1] ) return 0.0;
172 if( (s_center[1] - source.getRadius() ) > map_boundary_p2[1] ) return 0.0;
173
174 int nthreads = Utilities::GetNThreads();
175 PosType totals[nthreads];
176 std::vector<std::thread> thr;
177
178 size_t block = map.size()/nthreads;
179 for(int i = 0; i < nthreads ;++i){
180 thr.push_back(std::thread(&PixelMap<T>::addsource_<S>,this
181 ,i*block,std::min((i+1)*block-1,map.size()-1)
182 ,oversample,std::ref(source)
183 ,std::ref(totals[i])));
184 }
185
186 for(int ii=0;ii < nthreads;++ii) thr[ii].join();
187
188 PosType total =0;
189 for(int ii=0;ii < nthreads;++ii) total += totals[ii];
190
191 return total;
192 }

532 template <typename S>
533 void addsource_(size_t i1,size_t i2,int oversample,
534 S &source,
535 PosType &total){
536 PosType tmp_res = resolution*1.0/oversample;
537 PosType tmp = tmp_res*tmp_res;
538 PosType bl = resolution /2 - 0.5*tmp_res;
539 PosType y[2],x[2];
540
541 total = 0;
542
543 for(size_t index = i1 ;index <= i2; ++index){
544 find_position(y,index);
545 y[0] -= bl;
546 y[1] -= bl;
547 for(int i = 0 ; i < oversample ; ++i){
548 x[0] = y[0] + i*tmp_res;
549 for(int j=0; j < oversample;++j){
550 x[1] = y[1] + j*tmp_res;
551 map[index] += source.SurfaceBrightness(x)*tmp;
552 total += source.SurfaceBrightness(x)*tmp;
553 }
554 }
555 }
556 }
