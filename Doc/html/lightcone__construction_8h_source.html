94  void LightCone::select(Point_3d xo,Point_3d v,double Length,double rlow,double rhigh
95  ,T* begin
96  ,T* end
97  ,Utilities::LockableContainer<std::vector<T> > &incone
98  ,bool periodic_boundaries){
99 
100  if(begin == end) return;
101 
102  Point_3d dx,x;
103  double r2,xp;
104  Point_3d n;
105  double rhigh2 = rhigh*rhigh;
106  double rlow2 = rlow*rlow;
107  T b;
108 
109  v /= v.length();
110 
111  // standard coordinates so cone is always centered on the x-axis
112  Point_3d y_axis,z_axis;
113 
114  y_axis[0] = -v[1];
115  y_axis[1] = v[0];
116  y_axis[2] = 0;
117 
118  y_axis /= y_axis.length();
119  z_axis = v.cross(y_axis);
120 
121  //std::cout << v*y_axis << " " << v*z_axis << " " << y_axis*z_axis << std::endl;
122 
123  // we can do better than this !!!!
124  int n0[2] = {(int)((rhigh + xo[0])/Length),(int)((xo[0] - rhigh)/Length) - 1 };
125  int n1[2] = {(int)((rhigh + xo[1])/Length),(int)((xo[1] - rhigh)/Length) - 1 };
126  int n2[2] = {(int)((rhigh + xo[2])/Length),(int)((xo[2] - rhigh)/Length) - 1 };
127  if(!periodic_boundaries){
128  n0[0] = n0[1] = 0;
129  n1[0] = n1[1] = 0;
130  n2[0] = n2[1] = 0;
131  }
132  for(auto a = begin ; a != end ; ++a){
133  dx[0] = (*a).x[0] - xo[0];
134  dx[1] = (*a).x[1] - xo[1];
135  dx[2] = (*a).x[2] - xo[2];
136 
137  int count = 0;
138 
139  for(n[0] = n0[1]; n[0] <= n0[0] ; ++n[0]){
140  for(n[1] = n1[1]; n[1] <= n1[0] ; ++n[1]){
141  for(n[2] = n2[1]; n[2] <= n2[0] ; ++n[2]){
142 
143  x = dx + n*Length;
144  xp = x*v;
145  if(xp > 0){
146  r2 = x.length_sqr();
147 
148  if(r2 > rlow2 && r2 < rhigh2){
149  if( r2*sin_theta_sqrt > r2 - xp*xp){
150  //std::cout << n << " | " << xo << " | "
151  // << v << " | " << x << std::endl;
152 
153  b = *a;
154  //b.x = x;
155  // rotate to standard reference frame
156  b.x[0] = xp;
157  b.x[1] = y_axis*x;
158  b.x[2] = z_axis*x;
159 
160  //std::cout << "b = " << b.x << std::endl;
161 
162  incone.push_back(b);
163  ++count;
164  }
165  }
166  }
167  }
168  }
169  }
170  if(count > 1) std::cout << " Warning: Same halo appears " << count << " times in light-cone." << std::endl;
171  }
