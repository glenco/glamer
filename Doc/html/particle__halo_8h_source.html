240

242  min_mass = (pp[multimass*i].mass() < min_mass) ? pp[multimass*i].mass() : min_mass;

546 template<typename PType>
547 void LensHaloParticles<PType>::rotate_particles(PosType theta_x,PosType theta_y){
548 
549  if(theta_x == 0.0 && theta_y == 0.0) return;
550 
551  PosType coord[3][3];
552  PosType cx,cy,sx,sy;
553 
554  cx = cos(theta_x); sx = sin(theta_x);
555  cy = cos(theta_y); sy = sin(theta_y);
556 
557  coord[0][0] = cy; coord[1][0] = -sy*sx; coord[2][0] = cx;
558  coord[0][1] = 0; coord[1][1] = cx; coord[2][1] = sx;
559  coord[0][2] = -sy; coord[1][2] = -cy*sx; coord[2][2] = cy*cx;
560 
561  PosType tmp[3];
562  int j;
563  /* rotate particle positions */
564  for(size_t i=0;i<Npoints;++i){
565  for(j=0;j<3;++j) tmp[j]=0.0;
566  for(j=0;j<3;++j){
567  tmp[0] += coord[0][j]*pp[i][j];
568  tmp[1] += coord[1][j]*pp[i][j];
569  tmp[2] += coord[2][j]*pp[i][j];
570  }
571  for(j=0;j<3;++j) pp[i][j]=tmp[j];
572  }
573 }

575 template<typename PType>
576 void LensHaloParticles<PType>::calculate_smoothing(int Nsmooth,PType *pp
577  ,size_t Npoints
578  ,bool verbose
579  ){
580 
581  int nthreads = Utilities::GetNThreads();
582 
583  if(verbose) std::cout << "Calculating smoothing of particles ..." << std::endl
584  << Nsmooth << " neighbors. If there are a lot of particles this could take a while." << std::endl;
585 
586  time_t to,t;
587  time(&to);
588 
589  // make 3d tree of particle postions
590  TreeSimple<PType> tree3d(pp,Npoints,2*Nsmooth,3,true);
591  // find distance to nth neighbour for every particle
592  if(Npoints < 1000){
593  //IndexType neighbors[Nsmooth];
594  for(size_t i=0;i<Npoints;++i){
595  pp[i].Size = tree3d.NNDistance(&pp[i][0],Nsmooth + 1);
596  }
597  }else{
598  size_t chunksize = Npoints/nthreads;
599  std::vector<std::thread> thr(nthreads);
600 
601  size_t N;
602  for(int ii = 0; ii < nthreads ;++ii){
603  if(ii == nthreads - 1){
604  N = Npoints - ii*chunksize;
605  }else N = chunksize;
606 
607  thr[ii] = std::thread(LensHaloParticles<PType>::smooth_,&tree3d
608  ,&(pp[ii*chunksize]),N,Nsmooth);
609  }
610  for(int ii = 0; ii < nthreads ;++ii) thr[ii].join();
611  }
612  time(&t);
613  if(verbose) std::cout << "done in " << difftime(t,to) << " secs" << std::endl;
614 }

657 template<typename PType>
658 void LensHaloParticles<PType>::makeSIE(
659  std::string new_filename
660  ,PosType redshift
661  ,double particle_mass
662  ,double total_mass
663  ,double sigma
664  ,double q
665  ,Utilities::RandomNumbers_NR &ran
666  ){
667 
668  size_t Npoints = total_mass/particle_mass;
669  PosType Rmax = (1+redshift)*total_mass*Grav*lightspeed*lightspeed/sigma/sigma/2;
670  Point_3d point;
671  double qq = sqrt(q);
672 
673  std::ofstream datafile;
674  datafile.open(new_filename);
675 
676  datafile << "# nparticles " << Npoints << std::endl;
677  datafile << "# mass " << particle_mass << std::endl;
678  // create particles
679  for(size_t i=0; i< Npoints ;++i){
680  point[0] = ran.gauss();
681  point[1] = ran.gauss();
682  point[2] = ran.gauss();
683 
684  point *= Rmax*ran()/point.length();
685 
686  point[0] *= qq;
687  point[1] /= qq;
688 
689  datafile << point[0] << " " << point[1] << " "
690  << point[2] << " " << std::endl;
691 
692  }
693 
694  datafile.close();
