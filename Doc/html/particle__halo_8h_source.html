509 template<typename PType>
510 void LensHaloParticles<PType>::rotate_particles(PosType theta_x,PosType theta_y){
511 
512  if(theta_x == 0.0 && theta_y == 0.0) return;
513 
514  PosType coord[3][3];
515  PosType cx,cy,sx,sy;
516 
517  cx = cos(theta_x); sx = sin(theta_x);
518  cy = cos(theta_y); sy = sin(theta_y);
519 
520  coord[0][0] = cy; coord[1][0] = -sy*sx; coord[2][0] = cx;
521  coord[0][1] = 0; coord[1][1] = cx; coord[2][1] = sx;
522  coord[0][2] = -sy; coord[1][2] = -cy*sx; coord[2][2] = cy*cx;
523 
524  PosType tmp[3];
525  int j;
526  /* rotate particle positions */
527  for(size_t i=0;i<Npoints;++i){
528  for(j=0;j<3;++j) tmp[j]=0.0;
529  for(j=0;j<3;++j){
530  tmp[0]+=coord[0][j]*pp[i][j];
531  tmp[1]+=coord[1][j]*pp[i][j];
532  tmp[2]+=coord[2][j]*pp[i][j];
533  }
534  for(j=0;j<3;++j) pp[i][j]=tmp[j];
535  }
536 }

540  ,size_t Npoints){
541 
542  int nthreads = Utilities::GetNThreads();
543 
544  std::cout << "Calculating smoothing of particles ..." << std::endl

546 
547  time_t to,t;
548  time(&to);
549 
550  // make 3d tree of particle postions
551  TreeSimple<PType> tree3d(pp,Npoints,2*Nsmooth,3,true);
552  // find distance to nth neighbour for every particle
553  if(Npoints < 1000){
554  //IndexType neighbors[Nsmooth];
555  for(size_t i=0;i<Npoints;++i){
556  pp[i].Size = tree3d.NNDistance(&pp[i][0],Nsmooth + 1);
557  }
558  }else{
559  size_t chunksize = Npoints/nthreads;
560  std::thread thr[nthreads];
561 
562  size_t N;
563  for(int ii = 0; ii < nthreads ;++ii){
564  if(ii == nthreads - 1){
565  N = Npoints - ii*chunksize;
566  }else N = chunksize;
567 
568  thr[ii] = std::thread(LensHaloParticles<PType>::smooth_,&tree3d
569  ,&(pp[ii*chunksize]),N,Nsmooth);
570  }
571  for(int ii = 0; ii < nthreads ;++ii) thr[ii].join();
572  }
573  time(&t);
574  std::cout << "done in " << difftime(t,to) << " secs" << std::endl;

618 template<typename PType>
619 void LensHaloParticles<PType>::makeSIE(
620  std::string new_filename
621  ,PosType redshift
622  ,double particle_mass
623  ,double total_mass
624  ,double sigma
625  ,double q
626  ,Utilities::RandomNumbers_NR &ran
627  ){
628 
629  size_t Npoints = total_mass/particle_mass;
630  PosType Rmax = (1+redshift)*total_mass*Grav*lightspeed*lightspeed/sigma/sigma/2;
631  Point_3d point;

641  point[0] = ran.gauss();
642  point[1] = ran.gauss();
643  point[2] = ran.gauss();
644 
645  point *= Rmax*ran()/point.length();
646 
647  point[0] *= qq;
648  point[1] /= qq;
649 
650  datafile << point[0] << " " << point[1] << " "
651  << point[2] << " " << std::endl;
