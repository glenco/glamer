198 template<typename Number>
199 bool InputParams::get(std::string label, Number& value) const
200 {
201  const_iterator it = params.find(label);
202 
203  if(it == params.end())
204  {
205  value = Number();
206  return false;
207  }
208 
209  std::istringstream sstr(it->second);
210 
211  sstr >> value;
212  if(sstr.fail())
213  {
214  std::cout << "Expected a numerical value for " << it->first << " in parameter file " << paramfile_name << std::endl;
215  return false;
216  }
217 
218  use_counter.use(it->first);
219 
220  return true;
221 }
