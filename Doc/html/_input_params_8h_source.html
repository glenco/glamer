196 template<typename Number>
197 bool InputParams::get(std::string label, Number& value) const
198 {
199  const_iterator it = params.find(label);
200 
201  if(it == params.end())
202  {
203  value = Number();
204  return false;
205  }
206 
207  std::istringstream sstr(it->second);
208 
209  sstr >> value;
210  if(sstr.fail())
211  {
212  std::cout << "Expected a numerical value for " << it->first << " in parameter file " << paramfile_name << std::endl;
213  return false;
214  }
215 
216  use_counter.use(it->first);
217 
218  return true;
219 }
