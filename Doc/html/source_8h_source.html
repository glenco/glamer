305  void pepper(int n,double s,Utilities::RandomNumbers_NR &ran){
306 
307  double f = SurfaceBrightness(source_x.x);
308 
309  double t=0.9,t2=0.3;
310 
311  for(int i=0 ; i < n ; ++i){
312  double r = source_r*sqrt(ran());
313  double theta = 2*PI*ran();
314  double ff = f*( t - t2 * ran() );
315 
316  Point_2d x( r*cos(theta) , r*sin(theta) );
317 
318  grains.emplace_back(ff,s,x);
319  }
320  }
