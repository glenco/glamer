424 void LensMap::ProcessFFTs(
425  float zerosize
426  ,T Wphi_of_k
427  ,my_fftw_plan &plan_padded // the plan must be initialized with thr right size arrays
428  ,bool do_alpha){
429 
430  assert(surface_density.size() == nx*ny);
431 
432  // size of the new map in x and y directions, factor by which each size is increased
433  int Nnx=int(zerosize*nx);
434  int Nny=int(zerosize*ny);
435  size_t NN = Nnx*Nny;
436  double boxlx = boxlMpc*zerosize;
437  double boxly = ny*boxlMpc*zerosize/nx;
438 
439  assert( (plan_padded.nx == Nnx)*(plan_padded.ny == Nny) );
440 
441  int imin = (Nnx-nx)/2;
442  int imax = (Nnx+nx)/2;
443  int jmin = (Nny-ny)/2;
444  int jmax = (Nny+ny)/2;
445 
446  size_t Nkx = (Nnx/2+1);
447 
448  double tmp = 2.*M_PI/boxlx;
449  std::vector<double> kxs(Nkx);
450  for( int i=0; i<Nkx; i++ ){
451  kxs[i] = i*tmp;
452  }
453  tmp = 2.*M_PI/boxly;
454  std::vector<double> kys(Nny);
455  for( int j=0; j<Nny; j++ ){
456  kys[j]=(j<Nny/2)?double(j):double(j-Nny);
457  kys[j] *= tmp;
458  }
459 
460  std::vector<double> extended_map( NN );
461 
462  // assume locate in a rectangular map and build up the new one
463  for( int j=0; j<Nny; j++ ){
464  for( int i=0; i<Nnx; i++ ){
465  if(i>=imin && i<imax && j>=jmin && j<jmax){
466  int ii = i-imin;
467  int jj = j-jmin;
468 
469  if(ii>=nx || jj>=ny){
470  std::cerr << " 1 error mapping " << ii << " " << jj << std::endl;
471  exit(1);
472  }
473  if(ii<0 || jj<0){
474  std::cerr << " 2 error mapping " << ii << " " << jj << std::endl;
475  exit(1);
476  }
477  assert(ii+nx*jj < surface_density.size());
478  assert(i+Nnx*j < extended_map.size());
479  extended_map[i+Nnx*j] = surface_density[ii+nx*jj];
480  //float tmp = extended_map[i+Nnx*j];
481  assert(!isnan(extended_map[i+Nnx*j]));
482  if(isinf(extended_map[i+Nnx*j])){
483  extended_map[i+Nnx*j] = 0;
484  }
485  }else{
486  extended_map[i+Nnx*j] = 0;
487  }
488 
489  }
490  }
491 
492  //std::vector<fftw_complex> fNmap(Nny*(Nnx/2+1));
493  //std::vector<fftw_complex> fphi( Nny*(Nnx/2+1) );
494  fftw_complex *fphi = new fftw_complex[Nny*Nkx];
495 
496  fftw_execute_dft_r2c(plan_padded.plan_r2c
497  ,extended_map.data(),fphi);
498 
499 
500  // fourier space
501  // std:: cout << " allocating fourier space maps " << std:: endl;
502 
503  // build modes for each pixel in the fourier space
504  for( int i=0; i<Nkx; i++ ){
505  for( int j=0; j<Nny; j++ ){
506 
507  double k2 = kxs[i]*kxs[i] + kys[j]*kys[j];
508  size_t k = i+(Nkx)*j;
509 
510  // null for k2 = 0 no divergence
511  if(k2 == 0){
512  fphi[k][0] = 0.;
513  fphi[k][1] = 0.;
514  }else{
515 
516  //assert(k < Nny*Nkx);
517  //assert(!isnan(fphi[k][0]));
518 
519  // fphi
520  //fphi[i+(Nkx)*j][0]= -2.*fNmap[i+(Nkx)*j][0]/k2;
521  //fphi[i+(Nkx)*j][1]= -2.*fNmap[i+(Nkx)*j][1]/k2;
522 
523  // fphi
524  fphi[k][0] *= -2./k2;
525  fphi[k][1] *= -2./k2;
526 
527  // apply window function
528  double w = Wphi_of_k(k2);
529  fphi[k][0] *= w;
530  fphi[k][1] *= w;
531  }
532  }
533  }
534

539  std::vector<double> realsp(Nnx*Nny);

617  int ii = i-imin;
618

620  }
621  }
622  }

628
