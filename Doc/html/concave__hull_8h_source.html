61  void convex_hull(std::vector<T> &P,std::vector<T> &hull_out)
62  {
63 
64  if(P.size() <= 3){
65  hull_out = P;
66  return;
67  }
68 
69  std::vector<T> hull;
70 
71  size_t n = P.size();
72  size_t k = 0;
73  hull.resize(2*n);
74 
75  // Sort points lexicographically
76  std::sort(P.begin(), P.end(),
77  [](T p1,T p2){
78  if(p1[0]==p2[0]) return p1[1] < p2[1];
79  return p1[0] < p2[0];});
80 
81 
82  // Build lower hull
83  for (size_t i = 0; i < n; i++) {
84  while (k >= 2 && crossD(hull[k-2], hull[k-1], P[i]) <= 0){
85  k--;
86  }
87  hull[k++] = P[i];
88  }
89 
90  // Build upper hull
91  for (long i = n-2, t = k+1; i >= 0; i--) {
92  while (k >= t && crossD(hull[k-2], hull[k-1], P[i]) <= 0){
93  k--;
94  assert(k > 1);
95  }
96  hull[k++] = P[i];
97  }
98 
99 
100  hull.resize(k);
101  hull.pop_back();
102 
103  std::swap(hull,hull_out);
104 
105  return;
106  }
