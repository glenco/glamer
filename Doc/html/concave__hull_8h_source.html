477 template<typename T>
478 void convex_hull(std::vector<T> &P,std::vector<T> &hull_out)
479 {
480 
481  if(P.size() <= 3){
482  hull_out = P;
483  return;
484  }
485 
486  std::vector<T> hull;
487 
488  size_t n = P.size();
489  size_t k = 0;
490  hull.resize(2*n);
491 
492  // Sort points lexicographically
493  std::sort(P.begin(), P.end(),
494  [](T p1,T p2){
495  if(p1[0]==p2[0]) return p1[1] < p2[1];
496  return p1[0] < p2[0];});
497 
498 
499  // Build lower hull
500  for (size_t i = 0; i < n; i++) {
501  while (k >= 2 && crossD(hull[k-2], hull[k-1], P[i]) <= 0){
502  k--;
503  }
504  hull[k++] = P[i];
505  }
506 
507  // Build upper hull
508  for (long i = n-2, t = k+1; i >= 0; i--) {
509  while (k >= t && crossD(hull[k-2], hull[k-1], P[i]) <= 0){
510  k--;
511  assert(k > 1);
512  }
513  hull[k++] = P[i];
514  }
515 
516 
517  hull.resize(k);
518  hull.pop_back();
519 
520  std::swap(hull,hull_out);
521 
522  return;
523 }

549  }

551  [&P](size_t i1,size_t i2){

553  return P[i1][0] < P[i2][0];});
554 
555

558  while (k >= 2 && crossD(P[hull[k-2]], P[hull[k-1]], P[sort_index[i]]) <= 0){

569  }
