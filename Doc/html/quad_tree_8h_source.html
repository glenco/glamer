264  /* cubic B-spline kernel for particle profile
265 
266  The lensing quantities are added to and a point mass is subtracted
267  */
268  inline void b_spline_profile(
269  PosType *xcm // vector in Mpc connecting ray to center of particle
270  ,PosType r // distance from center in Mpc
271  ,PosType Mass // mass in solar masses
272  ,PosType size // size scale in Mpc
273  ,PosType *alpha // deflection angle times Sigma_crit
274  ,KappaType *kappa // surface density
275  ,KappaType *gamma // shear times Sigma_crit
276  ,KappaType *phi
277  ) const {
278 
279  PosType q = r/size;
280  PosType M,sigma;
281  if(q > 2){
282  sigma = 0;
283  M = 1;
284  }else{
285  PosType q2=q*q,q3=q2*q,q4=q2*q2,q5=q4*q;
286 
287  sigma = (8 - 12*q + 6*q2 - q3)/4;
288  if(q > 1){
289  sigma *= 10/size/size/7/pi;
290  M = (-1 + 20*q2*(1 - q + 3*q2/8 - q3/20) )/7;
291  *phi += Mass*(-1232. + 1200*q2 - 800.*q3 + 225.*q4 - 24*q5 + 120*log(2./q) )/840/pi;
292  }else{
293  sigma = 10*( sigma - 1 + 3*q - 3*q2 + q3)/size/size/7/pi;
294  M = 10*q2*(1 - 3*q/4 + 3*q3/10)/7;
295 
296  *phi += Mass*( phiintconst + 10*(q2/2 - 3*q4/4 + 3*q5/50)/7
297  )/pi;
298  }
299  }
300 
301  PosType alpha_r,gt; // deflection * Sig_crit / Mass
302  alpha_r = (M-1)/pi/r;
303  gt = alpha_r/r - sigma;
304 
305  alpha[0] -= Mass*alpha_r*xcm[0]/r;
306  alpha[1] -= Mass*alpha_r*xcm[1]/r;
307  gamma[0] -= gt*Mass*(xcm[0]*xcm[0]-xcm[1]*xcm[1])/r/r;
308  gamma[1] -= gt*Mass*2*xcm[0]*xcm[1]/r/r;
309  *kappa += Mass*sigma;
310  *phi -= Mass*log(r)/pi;
311  }
